#!/usr/bin/python3

def test_add_integer():
    assert add_integer(3, 5) == 8
    assert add_integer(10, 20) == 30
    assert add_integer(-5, 5) == 0
    assert add_integer(3.5, 2) == 5
    assert add_integer(4, 7.8) == 11
    assert add_integer(3.5, 7.8) == 11
    assert add_integer(3.5) == 101
    assert add_integer() == 98
    try:
        add_integer("3", 5)
    except TypeError as e:
        assert str(e) == "a must be an integer or b must be an integer"
    try:
        add_integer(3, "5")
    except TypeError as e:
        assert str(e) == "a must be an integer or b must be an integer"
